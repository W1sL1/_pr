# Подключение модуля для линейной алгебры (матрицы, определители, собственные значения)
using LinearAlgebra

# Определение функции для анализа кривой второго порядка
function solve_conic()
    # Вывод заголовка и описания формата уравнения
    println("--- Анализ кривой второго порядка ---")
    println("Уравнение: Ax² + 2Bxy + Cy² + 2Dx + 2Ey + F = 0")
    
    # === Ввод коэффициентов ===
    # print() выводит текст без перевода строки
    # readline() считывает строку из консоли
    # parse(Float64, ...) преобразует строку в число с плавающей точкой
    print("Введите A: "); A = parse(Float64, readline())
    print("Введите B (половина коэффициента при xy): "); B = parse(Float64, readline())
    print("Введите C: "); C = parse(Float64, readline())
    print("Введите D (половина коэффициента при x): "); D = parse(Float64, readline())
    print("Введите E (половина коэффициента при y): "); E = parse(Float64, readline())
    print("Введите F: "); F = parse(Float64, readline())

    # === Формирование матриц ===
    # Квадратные скобки [] создают массив
    # Пробел разделяет элементы в строке, точка с запятой ; разделяет строки
    Matrix_small = [A B; B C]      # Матрица квадратичной формы 2×2
    Matrix_big = [A B D; B C E; D E F]  # Полная матрица кривой 3×3

    # === Вычисление инвариантов ===
    # tr() - след матрицы (сумма диагональных элементов)
    I1 = tr(Matrix_small)      # Первый инвариант
    I2 = det(Matrix_small)     # Второй инвариант (определитель)
    I3 = det(Matrix_big)       # Третий инвариант (определитель)

    # Вывод вычисленных инвариантов
    # $() - интерполяция переменных в строку
    println("\nИнварианты: I1 = $I1, I2 = $I2, I3 = $I3")

    # === Определение типа кривой по знаку второго инварианта ===
    if I2 > 0
        type = "Эллипс"
    elseif I2 < 0
        type = "Гипербола"
    else
        # При I2 = 0 кривая является параболой
        # Данный расчет не поддерживает параболы
        type = "Парабола (не поддерживается данным расчетом)"
        println("Тип: $type")
        return  # Выход из функции, так как парабола не обрабатывается
    end
    println("Тип кривой: $type")

    # === Нахождение собственных чисел и векторов ===
    # eigen() возвращает кортеж: (собственные значения, собственные векторы)
    vals, vecs = eigen(Matrix_small)
    println("Собственные числа: $(vals[1]) и $(vals[2])")
    # vecs[:,1] - первый столбец матрицы (первый собственный вектор)
    println("Собственные векторы: v1 = $(vecs[:,1]), v2 = $(vecs[:,2])")

    # === Нахождение центра кривой ===
    # Решение системы уравнений:
    # A*x + B*y + D = 0
    # B*x + C*y + E = 0
    # Оператор \ решает систему линейных уравнений (левое деление)
    center = Matrix_small \ [-D, -E]
    x0, y0 = center  # Множественное присваивание
    println("Центр: ($x0, $y0)")

    # === Приведение к каноническому виду ===
    # После поворота и переноса уравнение принимает вид:
    # λ1*X² + λ2*Y² + I3/I2 = 0
    # Отсюда: λ1*X² + λ2*Y² = -I3/I2
    rhs = -I3 / I2
    
    # Вычисление квадратов полуосей
    a_sq = rhs / vals[1]
    b_sq = rhs / vals[2]

    # Вычисление полуосей (берем модуль, так как для гиперболы значения могут быть отрицательными)
    a = sqrt(abs(a_sq))
    b = sqrt(abs(b_sq))
    
    # Сортировка полуосей в порядке убывания (a - большая/действительная)
    semiaxes = sort([a, b], rev=true)
    
    # round() округляет до указанного количества знаков
    println("Полуоси: a ≈ $(round(semiaxes[1], digits=4)), b ≈ $(round(semiaxes[2], digits=4))")

    # === Вычисление эксцентриситета ===
    if type == "Эллипс"
        # Для эллипса: e = √(1 - b²/a²), где a ≥ b
        e = sqrt(1 - (min(a,b)^2 / max(a,b)^2))
    else  # Гипербола
        # Для гиперболы: e = √(1 + b²/a²), где a - действительная полуось
        # Определяем, какая из осей является действительной
        # Если знаки собственного значения и правой части различны,
        # то действительная ось соответствует этому собственному значению
        if sign(vals[1]) != sign(rhs)
            e = sqrt(1 + abs(vals[2]/vals[1]))
        else
            e = sqrt(1 + abs(vals[1]/vals[2]))
        end
    end
    println("Эксцентриситет: e ≈ $(round(e, digits=4))")

    # === Ожидание нажатия Enter перед завершением ===
    println("\nНажмите Enter, чтобы выйти...")
    readline()  # Ожидание ввода от пользователя
end

# === Вызов функции для выполнения программы ===
solve_conic()
